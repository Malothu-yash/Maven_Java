Minikube (Nginx Service Deployment):
minikube start
kubectl create deployment nginx --image=nginx
kubectl expose deployment nginx --port=80 --target-port=80 --name=nginx-service --type=ClusterIP
kubectl port-forward svc/nginx-service 8090:80
# In browser open http://localhost:8090
# List pods and services in another terminal:
kubectl get pods -o wide
kubectl get svc
